FullSimulation = Faculty %>%
  filter(Title2 == "Full") %>%
  specify(Salary ~ Gender) %>%
  hypothesize(null = "independence") %>%
  generate(reps = 1000, type = "permute") %>%
  calculate(stat = "diff in means", order = c("Male","Female"))

Faculty %>%

  group_by(Gender) %>%
  summarise(
   mean(Salary)
